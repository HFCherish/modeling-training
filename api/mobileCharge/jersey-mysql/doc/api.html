Body

Type: application/json

Schema:

{
  "$schema": "http://json-schema.org/draft-04/schema#",
  "type": "object",
  "properties": {
    "id": {
      "type": "string"
    },
    "phone_number": {
      "type": "number"
    },
    "ID_Card": {
      "type": "string",
      "description": "certificate with real identity"
    },
    "card_locale": {
      "type": "string"
    },
    "links": {
      "type": "array",
      "items": {
        "type": "object",
        "properties": {
          "rel": {
            "type": "string"
          },
          "uri": {
            "type": "string"
          }
        },
        "required": [
          "rel",
          "uri"
        ]
      }
    }
  },
  "required": [
    "id",
    "phone_number",
    "ID_Card",
    "card_locale",
    "links"
  ]
}

Example:

{
	"id": "xxx",
	"phone_number": 1236879,
	"ID_Card": "234542",
	"card_locale": "beijing",
	"links": [{
		"rel": "self",

	}]
}
